--CHR 1 \
--geno test_.05_.2/test.geno \
--fam test_.05_.2/test.fam \
--bim test_.05_.2/test.bim \
--ref refs_sim.txt \
--admx admx_sim.txt \
--out test_.05_.2 \
-w 100 \
--overlap 50 \
--proc 5 \
--dr PCA \
--MSprint
